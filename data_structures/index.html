Data Structures

Arrays

Java

String cars[] = new String[] { "Toyota", "Mercedes", "BMW", "Volkswagen", "Skoda" };

System.out.println(cars[1]); // Mercedes

GoLang

cars := []string{"Toyota", "Mercedes", "BMW", "Volkswagen", "Skoda"}

fmt.Println(cars[1]) // Mercedes

Multidimensional array

Java

String langs[][] = new String[][]{{"C#", "C", "Python"},
                   {"Java", "Scala", "Perl"}, 
                   {"C++", "Go", "HTML"}}

GoLang

langs := [3][3]string{{"C#", "C", "Python"},
                   {"Java", "Scala", "Perl"},
                   {"C++", "Go", "HTML"}}

Lists

Java

List<String> letters = List.of("a", "b", "c", "d");

letters.add("e");

int length = letters.size();

GoLang

letters := []string{"a", "b", "c", "d"}

letters = append(letters, "e")

length := len(letters)

Maps

Java

Map<String, String> mapA = Map.of("key1", "element1");

mapA.put("key2", "element2");

mapA.remove("key2")

if (mapA.contains("key1")) {
    System.out.println(mapA.get("key1")); // element2
}

GoLang

mapA := map[string]string{"key1":"element1"}

mapA["key2"] = "element2"

delete(mapA, "key2")

if value, ok = mapA["key1"]; ok == true {
	mapA["key1"] // element2
}
